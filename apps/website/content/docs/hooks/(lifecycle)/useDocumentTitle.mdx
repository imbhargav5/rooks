## About

Sets the document title and can optionally restore the previous title on unmount.

[//]: # "Main"

## Examples

### Set a page title on mount

```jsx
import { useDocumentTitle } from "rooks";

function App() {
  useDocumentTitle("Project dashboard");

  return <h1>Project dashboard</h1>;
}

export default App;
```

### Restore the previous title when the component unmounts

```jsx
import { useState } from "react";
import { useDocumentTitle } from "rooks";

function PreviewPanel() {
  useDocumentTitle("Preview mode", { resetOnUnmount: true });

  return <p>Close this panel to restore the previous document title.</p>;
}

export default function App() {
  const [isOpen, setIsOpen] = useState(true);

  return (
    <div>
      <button onClick={() => setIsOpen((current) => !current)}>
        Toggle preview panel
      </button>
      {isOpen ? <PreviewPanel /> : <p>Preview closed.</p>}
    </div>
  );
}
```

## Arguments

| Argument  | Type                           | Description                 |
| --------- | ------------------------------ | --------------------------- |
| `title`   | `string`                       | The document title to apply |
| `options` | `{ resetOnUnmount?: boolean }` | Optional behavior flags     |

## Return

`void`
